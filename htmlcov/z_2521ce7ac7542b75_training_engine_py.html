324 def _save_checkpoint(self, epoch: int, metrics: EpochMetrics) -> None: 

325 """Save training checkpoint""" 

326 checkpoint_path = self.checkpoint_dir / f"checkpoint_epoch_{epoch}.json" 

327 checkpoint_data = { 

328 'epoch': epoch, 

329 'metrics': vars(metrics), 

330 'config': vars(self.config) 

331 } 

332 

333 with open(checkpoint_path, 'w') as f: 

334 json.dump(checkpoint_data, f, indent=2, default=str)

336 def _load_checkpoint(self, checkpoint_path: Path) -> int: 

337 """Load checkpoint and return starting epoch""" 

338 checkpoint_path = Path(checkpoint_path) 

339 

340 if not checkpoint_path.exists(): 

341 print(f"Checkpoint not found: {checkpoint_path}") 

342 return 0 

343 

344 with open(checkpoint_path, 'r') as f: 

345 checkpoint_data = json.load(f) 

346 

347 return checkpoint_data.get('epoch', 0)
